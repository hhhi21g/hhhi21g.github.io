```c++
#include <vector>
#include <unordered_map>  // 无序哈希表
using namespace std;

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> mp;  // 值->下标的映射

        for(int i = 0;i<nums.size();i++){
            int need = target - nums[i];
            if(mp.count(need)){  // 判断need是否存在
                return {mp[need],i};
            }
            mp[nums[i]] = i;
        }  
        return {}; 
    }
};
```

定义：dp[i] = 以num[i]结尾的最大子数组和

两种选择：

1. 单独开始：nums[i]

2. 接在前面：dp[i-1] + nums[i]

所以：

```c++
dp[i] = max(nums[i],dp[i-1]+nums[i])
```

#### (1) 反转链表

```c++
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* pre = nullptr;
        ListNode* cur = head;
        ListNode* next;
        while (cur != nullptr) {
            next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }
};
```

#### (2) 判断环形

```c++
// 快慢指针：
// 快指针：一次移动2；
// 慢指针：一次移动1.
// 若存在环形，快慢指针一定会相遇
class Solution {
public:
    bool hasCycle(ListNode*head) {
        ListNode*fast = head;
        ListNode*slow = head;
        while(fast && fast->next && fast->next->next){
            fast = fast->next->next;
            slow = slow->next;
            if(fast == slow)
                return 1;
        }
        return 0;
    }
};
```

### 5. 滑动窗口

维护一个动态区间(窗口)，通常用两个指针表示；通过移动两个指针来达到目的，而不是暴力枚举所有区间。

**例：无重复字符的最长子序列**

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char>window;
        int left = 0, right = 0;
        int res = 0;
        for (int right = 0; right < s.size(); right++) {
            while (window.count(s[right])) {
                window.erase(s[left]);
                left++;
            }
            window.insert(s[right]);
            res = max(res, (right - left + 1));
        }

        return res;
    }
};
```
