```c++
#include <vector>

using namespace std;

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> mp;  // 值->下标的映射

        for(int i = 0;i<nums.size();i++){
            int need = target - nums[i];
            if(mp.count(need)){  // 判断need是否存在
                return {mp[need],i};
            }
            mp[nums[i]] = i;
        }  
        return {}; 
    }
};
```

```c++
// 快慢指针：
// 快指针：一次移动2；
// 慢指针：一次移动1.
// 若存在环形，快慢指针一定会相遇
class Solution {
public:
    bool hasCycle(ListNode*head) {
        ListNode*fast = head;
        ListNode*slow = head;
        while(fast && fast->next && fast->next->next){
            fast = fast->next->next;
            slow = slow->next;
            if(fast == slow)
                return 1;
        }
        return 0;
    }
};
```

#### (3) 排序链表

> 普通冒泡排序可能会超时

```sql
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (head==nullptr || head->next==nullptr)return head;
        // fast从head->next开始，否则对于两个节点的链表会陷入死循环
        ListNode* fast = head->next, * slow = head;
		
        // 通过快慢指针查找链表中点
        while (fast && fast->next) {
            slow = slow->next;
            fast = fast->next->next;
        }

        ListNode* r = slow->next;
        slow->next = nullptr;

        ListNode* lList = sortList(head);
        ListNode* rList = sortList(r);

        return merge(lList, rList);
    }

    ListNode* merge(ListNode* l, ListNode* r) {
        if (!l)return r;
        if (!r)return l;
        ListNode* res = new ListNode(0);
        ListNode* tmp = res;
        while (l && r) {
            if (l->val > r->val) {
                tmp->next = r;
                r = r->next;
            }
            else {
                tmp->next = l;
                l = l->next;
            }
            tmp = tmp->next;
        }

        if (!l)
            tmp->next = r;
        else
            tmp->next = l;

        return res->next;
    }
};
```

### 5. 滑动窗口

维护一个动态区间(窗口)，通常用两个指针表示；通过移动两个指针来达到目的，而不是暴力枚举所有区间。

**例：无重复字符的最长子序列**

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char>window;
        int left = 0, right = 0;
        int res = 0;
        for (int right = 0; right < s.size(); right++) {
            while (window.count(s[right])) {
                window.erase(s[left]);
                left++;
            }
            window.insert(s[right]);
            res = max(res, (right - left + 1));
        }

        return res;
    }
};
```

**例：滑动窗口最大值**，通过维护单调递减队列处理，存储下标而不是值

```c++
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {

        vector<int> res;
        deque<int>dq;

        for (int i = 0; i < nums.size(); i++) {  // 不在队列内
            while (dq.size() != 0 && dq.front() < i - k + 1) {
                dq.pop_front();
            }

            while (dq.size() != 0 && nums[dq.back()] < nums[i]) {  // 单调
                dq.pop_back();
            }

            dq.push_back(i);

            if (i > k - 2)
                res.push_back(nums[dq.front()]);
        }
        
        return res;
    }
};
```

### 6. 前缀和

**例：数组中和为k的子数组的个数**

```c++
pre:到下标i处的前缀和，即n1+n2+...+ni;
假设：ni+...+nj = k
则：pre(j) - pre(i-1) = k
即: pre(j) - k = pre(i-1)
结论：若能够找到一个已有前缀和，使得该前缀和=当前前缀和-k，则已有前缀和出现的次数为当前前缀和能够产生的子数组个数。
```

```c++
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<int, int> pre_cnt; // 前缀和->出现次数
        int pre = 0;
        int res = 0;
        pre_cnt[0] = 1;
        for (int i = 0; i < nums.size(); i++) {
            pre += nums[i];
            if (pre_cnt.find(pre - k) != pre_cnt.end()) {
                res += pre_cnt[pre - k];
            }
            pre_cnt[pre]++;
        }
        return res;
    }
};
```

### 7. 深拷贝

**例：随机链表的深拷贝**, 复制为独立的两份，每个节点地址均不同

```c++
class Solution {
public:
    Node* copyRandomList(Node* head) {
        unordered_map<Node*, Node*> mp;
        for (Node* tmp = head; tmp != nullptr; tmp = tmp->next) {
            mp[tmp] = new Node(tmp->val);
        }

        for (Node* tmp = head; tmp != nullptr; tmp = tmp->next) {
            if (tmp->next)
                mp[tmp]->next = mp[tmp->next];
            else
                mp[tmp]->next = nullptr;

            if (tmp->random)
                mp[tmp]->random = mp[tmp->random];
            else
                mp[tmp]->random = nullptr;
        }
        return mp[head];
    }
};
```

#### (2) 层级遍历

```c++
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {

        vector<vector<int>> res;
        if (!root)return res;
        
        deque<TreeNode*> head;
        head.push_back(root);
        int size;

        while (head.size()!=0) {
            size = head.size();
            vector<int> level;

            for (int i = 0; i < size; i++) {

                TreeNode* rootnow = head.front();
                head.pop_front();
                level.push_back(rootnow->val);

                if (rootnow->left) {
                    head.push_back(rootnow->left);
                }
                if (rootnow->right) {
                    head.push_back(rootnow->right);
                }
            }
            res.push_back(level);
        }
        return res;
    }
};
```

#### (5) 对称二叉树

最初思路：前序遍历和后序遍历一致，即为对称

**当二叉树所有节点数值相同时不起效**

递归处理：

```sql
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        return midRec(root->left, root->right);
    }

    bool midRec(TreeNode* l, TreeNode* r) {
        if (l == nullptr && r == nullptr)return true;
        if ((l && !r) || (!l && r))return false;
        if (l->val != r->val)return false;
        return midRec(l->left, r->right) && midRec(l->right, r->left);
    }
};
```

#### (6) 数组->平衡二叉树

```sql
class Solution {
public:
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        TreeNode*res = build(nums,0,nums.size()-1);
        return res;
    }

    TreeNode*build(vector<int>&nums,int left,int right){
        if(left>right) return nullptr;

        int mid = (left+right)/2;
        TreeNode*res = new TreeNode(nums[mid]);

        res->left = build(nums,left,mid-1);
        res->right = build(nums,mid+1,right);
        return res;
    }
};
```

### 9. 二分查找

模板：

```sql
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int l = 0, r = nums.size() - 1;
        while (l <= r) {
            int mid = (r + l) / 2;

            if (nums[r] < target)
                return r + 1;
            if (nums[l] > target)
                return l;

            if (nums[mid] == target)
                return mid;
            else if (nums[mid] < target)
                l = mid + 1;
            else if (nums[mid] > target)
                r = mid - 1;
        }
        return -1;
    }
};

```
