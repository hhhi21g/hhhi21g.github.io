```sql
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (head==nullptr || head->next==nullptr)return head;
        // fast从head->next开始，否则对于两个节点的链表会陷入死循环
        ListNode* fast = head->next, * slow = head;
		
        // 通过快慢指针查找链表中点
        while (fast && fast->next) {
            slow = slow->next;
            fast = fast->next->next;
        }

        ListNode* r = slow->next;
        slow->next = nullptr;

        ListNode* lList = sortList(head);
        ListNode* rList = sortList(r);

        return merge(lList, rList);
    }

    ListNode* merge(ListNode* l, ListNode* r) {
        if (!l)return r;
        if (!r)return l;
        ListNode* res = new ListNode(0);
        ListNode* tmp = res;
        while (l && r) {
            if (l->val > r->val) {
                tmp->next = r;
                r = r->next;
            }
            else {
                tmp->next = l;
                l = l->next;
            }
            tmp = tmp->next;
        }

        if (!l)
            tmp->next = r;
        else
            tmp->next = l;

        return res->next;
    }
};
```

### 5. 滑动窗口

维护一个动态区间(窗口)，通常用两个指针表示；通过移动两个指针来达到目的，而不是暴力枚举所有区间。

**例：无重复字符的最长子序列**

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char>window;
        int left = 0, right = 0;
        int res = 0;
        for (int right = 0; right < s.size(); right++) {
            while (window.count(s[right])) {
                window.erase(s[left]);
                left++;
            }
            window.insert(s[right]);
            res = max(res, (right - left + 1));
        }

        return res;
    }
};
```

```c++
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {

        vector<int> res;
        deque<int>dq;

        for (int i = 0; i < nums.size(); i++) {  // 不在队列内
            while (dq.size() != 0 && dq.front() < i - k + 1) {
                dq.pop_front();

            dq.push_back(i);

            if (i > k - 2)
                res.push_back(nums[dq.front()]);
```

### 6. 前缀和

**例：数组中和为k的子数组的个数**

```c++
pre:到下标i处的前缀和，即n1+n2+...+ni;
假设：ni+...+nj = k
则：pre(j) - pre(i-1) = k
即: pre(j) - k = pre(i-1)
结论：若能够找到一个已有前缀和，使得该前缀和=当前前缀和-k，则已有前缀和出现的次数为当前前缀和能够产生的子数组个数。
```

```c++
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<int, int> pre_cnt; // 前缀和->出现次数
        int pre = 0;
        int res = 0;
        pre_cnt[0] = 1;
        for (int i = 0; i < nums.size(); i++) {
            pre += nums[i];
            if (pre_cnt.find(pre - k) != pre_cnt.end()) {
                res += pre_cnt[pre - k];
            }
            pre_cnt[pre]++;
        }
        return res;
    }
};
```

```c++
class Solution {
```

```c++
    vector<vector<int>> levelOrder(TreeNode* root) {

        vector<vector<int>> res;
        if (!root)return res;
        
        deque<TreeNode*> head;
        head.push_back(root);
        int size;

        while (head.size()!=0) {

                    head.push_back(rootnow->left);
                }
                if (rootnow->right) {
                    head.push_back(rootnow->right);
                }
            }
            res.push_back(level);
        }
        return res;
    }
};
```

```c++
        if (l == nullptr && r == nullptr)return true;
```

```c++
class Solution {
public:
    unordered_map<int, int> mp;
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        for (int i = 0; i < inorder.size(); i++) {
            mp[inorder[i]] = i;
        }

        return buildTreeNow(preorder, 0, preorder.size() - 1,inorder, 0, inorder.size() - 1);

    }

    TreeNode* buildTreeNow(vector<int>&preorder, int lpre, int rpre,
        vector<int>&inorder, int lin, int rin) {
        if (lpre > rpre || lin > rin)return nullptr;
        TreeNode* root = new TreeNode(preorder[lpre]);
        int lidx = mp[root->val];
        int lsize = lidx - lin;

        root->left = buildTreeNow(preorder, lpre + 1, lpre + lsize,
            inorder, lin, lidx - 1);

        root->right = buildTreeNow(preorder, lpre + lsize + 1, rpre,
            inorder, lidx + 1, rin);

        return root;
    }
};
```

#### (2) BFS

类似二叉树的层级遍历。**注意：**deque<pair<int,int>>速度和内存远优于deque<vector\<int>>

```c++

class Solution {
public:
    deque<pair<int,int>> bfsq;
    int res = 0;
    int orangesRotting(vector<vector<int>>& grid) {
        int fresh = 0;
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[0].size(); j++) {
                if (grid[i][j] == 2) {
                    bfsq.push_back({ i,j});  // 先遍历整个网格，将所有腐烂加入队列，多源bfs
                }
                if (grid[i][j] == 1)
                    fresh++;

            }
        }
        if (fresh == 0)return 0;

        bfs(grid, fresh);

        if (fresh != 0)return -1;
        return res;
    }

    void bfs(vector<vector<int>> &grid, int &fresh) {
        while (bfsq.size() != 0) {
            int n = bfsq.size();
            bool flag = false;

            for (int k = 0; k < n; k++) {
                int i = bfsq.front().first;
                int j = bfsq.front().second;
                bfsq.pop_front();

                if (i - 1 >= 0 && grid[i - 1][j] == 1) {
                    grid[i - 1][j] = 2;
                    bfsq.push_back({ i - 1, j });
                    fresh--;
                    flag = true;
                }
                if (i + 1 < grid.size() && grid[i + 1][j] == 1) {
                    grid[i + 1][j] = 2;
                    bfsq.push_back({ i + 1, j });
                    fresh--;
                    flag = true;
                }
                if (j - 1 >= 0 && grid[i][j - 1] == 1) {
                    grid[i][j - 1] = 2;
                    bfsq.push_back({ i, j - 1 });
                    fresh--;
                    flag = true;
                }
                if (j + 1 < grid[0].size() && grid[i][j + 1] == 1) {
                    grid[i][j + 1] = 2;
                    bfsq.push_back({ i, j + 1 });
                    fresh--;
                    flag = true;
                }
            }

            if (flag) res++;
            if (fresh == 0)return;
        }
    }
};
```

```c++
class Solution {

        for (int i = 0; i < numCourses; i++) {
            if (indegree[i] == 0) {
                q.push_back(i);
                learn++;
            }
        }

        while (q.size() != 0) {
            int head = q.front();
            q.pop_front();
            for (int i = 0; i < graph[head].size(); i++) {
                indegree[graph[head][i]]--;
                // 前置课程均已学习完成，可加入待学习队列
                if (indegree[graph[head][i]] == 0) {
                    q.push_back(graph[head][i]);
                    learn++;
                }
            }
        }

        if (learn == numCourses)return true;
        return false;
    }
};
```

dfs内部是否有循环：往下走一步是否有多种选择，若下一步可以任意选，则加循环；若只是上下左右这种，可用if代替

```c++
class Solution {
public:
    vector<vector<int>> res;
    vector<vector<int>> permute(vector<int>& nums) {
        vector<int>path;
        vector<int>used(nums.size(),0);
        dfs(nums, path, used);
        return res;
    }

    void dfs(vector<int>& nums, vector<int>path, vector<int>used) {
        if (path.size() == nums.size()) {
            res.push_back(path);
            return;
        }

        for (int i = 0; i < nums.size(); i++) {
            if (used[i])continue;
            path.push_back(nums[i]);
            used[i] = true;

            dfs(nums, path, used);

            path.pop_back();
            used[i] = false;
        }
    }
};
```

##### b. 子集

注意区别：

- depth: 用于记录当前递归层，避免每次都从nums[0]开始，极大的节约了时间；
- used不需要再置零，使用过的数字不会再次进行尝试。

```c++
class Solution {
public:
    vector<vector<int>> res;
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<int>path;
        vector<int>used(nums.size(), 0);

        res.push_back(path);
        dfs(nums, path, used);
        
        return res;
    }

    int depth = 0;
    void dfs(vector<int>& nums, vector<int>& path, vector<int> used) {
        for (int i = depth; i < nums.size(); i++) {
            if (used[i] == 1)continue;
            path.push_back(nums[i]);
            used[i]++;

            res.push_back(path);

            depth++;
            dfs(nums, path, used);
            depth--;

            path.pop_back();
        }
        return;
    }
};
```

### 11. 堆

**例：返回第k大的数**

先建大顶堆，再将最大元素放到最后，对其他元素重新建堆。

```sql
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        int n = nums.size();

        for (int i = n / 2 - 1; i >= 0; i--) {
            heap(nums, n, i);
        }

        for (int i = n - 1; i >= n - k + 1; i--) {
            swap(nums[0], nums[i]);
            heap(nums, i, 0);
        }

        return nums[0];
    }

    void heap(vector<int>& nums, int n, int i) {
        int largest = i;
        int l = 2 * i + 1;
        int r = 2 * i + 2;

        if (l<n && nums[l]>nums[largest])
            largest = l;
        if (r<n && nums[r]>nums[largest])
            largest = r;

        if (largest != i) {
            swap(nums[i], nums[largest]);
            heap(nums, n, largest);
        }
    }
};
```

#### (3) 覆盖问题

```c++
nums[i] = 从i最多能跳多远
```

a. 是否能跳到最后？

**维护当前能到达的最远位置**

> nums表示可以跳跃的最大长度，因此只需最远位置能够到达最后一个下标即可，没有跳过去而不能到达的风险

```c++
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int maxReach = 0;
        for(int i = 0;i<nums.size();i++){
            if(i>maxReach)
                return false;
            maxReach = max(maxReach,i+nums[i]);
        }
        return true;
    }
};
```

b. 跳到最后最少要多少步？

**维护每一次跳跃可以到达的最大位置**

```c++
class Solution {
public:
    int jump(vector<int>& nums) {
        int end = 0;
        int distance = 0;
        int step = 0;

        for (int i = 0; i < nums.size(); i++) {
            distance = max(distance, i + nums[i]);

            if (i == end && end != nums.size()-1) {
                step++;
                end = distance;
            }
        }
        return step;
    }
};
```

#### (4) 买卖股票

```c++
prices[i] = 第 i 天价格
```

a. 只能买一次、卖一次

```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int Min_Profile = INT_MAX;
        int res = 0;
        for (int i = 0; i < prices.size(); i++) {
            Min_Profile = min(Min_Profile, prices[i]);  // 到目前为止的最低价
            res = max(res, prices[i] - Min_Profile);
        }
        return res;
    }
};
```

### 13. 动态规划

```c++
// 模板：楼梯问题
class Solution {
public:
    int res = 0;
    int climbStairs(int n) {
        if (n == 1 )return 1;
        vector<int> dp(n + 1);
        dp[1] = 1;
        dp[2] = 2;
        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }
};
```

```c++
// 以零钱问题为例：用时优化写法
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        vector<int> dp(amount + 1, amount+1);

        dp[0] = 0;
        sort(coins.begin(), coins.end());

        for (int i = 1; i <= amount; i++) {
            for (int coin : coins) {
                if (coin > i)break;

                dp[i] = min(dp[i], dp[i - coin] + 1);
            }
        }

        return dp[amount] == amount + 1 ? -1 : dp[amount];
    }
};
```
