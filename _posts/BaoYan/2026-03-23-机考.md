```c++
#include <vector>
#include <unordered_map>  // 无序哈希表
using namespace std;

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> mp;  // 值->下标的映射

        for(int i = 0;i<nums.size();i++){
            int need = target - nums[i];
            if(mp.count(need)){  // 判断need是否存在
                return {mp[need],i};
            }
            mp[nums[i]] = i;
        }  
        return {}; 
    }
};
```

```c++
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* pre = nullptr;
        ListNode* cur = head;
        ListNode* next;
        while (cur != nullptr) {
            next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }
};
```

#### (2) 判断环形

```c++
// 快慢指针：
// 快指针：一次移动2；
// 慢指针：一次移动1.
// 若存在环形，快慢指针一定会相遇
class Solution {
public:
    bool hasCycle(ListNode*head) {
        ListNode*fast = head;
        ListNode*slow = head;
        while(fast && fast->next && fast->next->next){
            fast = fast->next->next;
            slow = slow->next;
            if(fast == slow)
                return 1;
        }
        return 0;
    }
};
```

### 5. 滑动窗口

维护一个动态区间(窗口)，通常用两个指针表示；通过移动两个指针来达到目的，而不是暴力枚举所有区间。

**例：无重复字符的最长子序列**

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char>window;
        int left = 0, right = 0;
        int res = 0;
        for (int right = 0; right < s.size(); right++) {
            while (window.count(s[right])) {
                window.erase(s[left]);
                left++;
            }
            window.insert(s[right]);
            res = max(res, (right - left + 1));
        }

        return res;
    }
};
```

**例：滑动窗口最大值**，通过维护单调递减队列处理，存储下标而不是值

```c++
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {

        vector<int> res;
        deque<int>dq;

        for (int i = 0; i < nums.size(); i++) {  // 不在队列内
            while (dq.size() != 0 && dq.front() < i - k + 1) {
                dq.pop_front();
            }

            while (dq.size() != 0 && nums[dq.back()] < nums[i]) {  // 单调
                dq.pop_back();
            }

            dq.push_back(i);

            if (i > k - 2)
                res.push_back(nums[dq.front()]);
        }
        
        return res;
    }
};
```

### 6. 前缀和

**例：数组中和为k的子数组的个数**

```c++
pre:到下标i处的前缀和，即n1+n2+...+ni;
假设：ni+...+nj = k
则：pre(j) - pre(i-1) = k
即: pre(j) - k = pre(i-1)
结论：若能够找到一个已有前缀和，使得该前缀和=当前前缀和-k，则已有前缀和出现的次数为当前前缀和能够产生的子数组个数。
```

```c++
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<int, int> pre_cnt; // 前缀和->出现次数
        int pre = 0;
        int res = 0;
        pre_cnt[0] = 1;
        for (int i = 0; i < nums.size(); i++) {
            pre += nums[i];
            if (pre_cnt.find(pre - k) != pre_cnt.end()) {
                res += pre_cnt[pre - k];
            }
            pre_cnt[pre]++;
        }
        return res;
    }
};
```

### 7. 深拷贝

**例：随机链表的深拷贝**, 复制为独立的两份，每个节点地址均不同

```c++
class Solution {
public:
    Node* copyRandomList(Node* head) {
        unordered_map<Node*, Node*> mp;
        for (Node* tmp = head; tmp != nullptr; tmp = tmp->next) {
            mp[tmp] = new Node(tmp->val);
        }

        for (Node* tmp = head; tmp != nullptr; tmp = tmp->next) {
            if (tmp->next)
                mp[tmp]->next = mp[tmp->next];
            else
                mp[tmp]->next = nullptr;

            if (tmp->random)
                mp[tmp]->random = mp[tmp->random];
            else
                mp[tmp]->random = nullptr;
        }
        return mp[head];
    }
};
```
